/**
 * 定义 {@link EventDispatcher} 派发事件的阶段的常量。
 * @enum {Number}
 * @since 1.0.0
 */
export default class EventPhase {
    /**
     * 定义事件的初始状态。
     * @readonly
     * @type {Number}
     * @since 1.0.0
     */
    static NONE = 0;
    
    /**
     * 定义事件处于捕获阶段。
     * @readonly
     * @type {Number}
     * @since 1.0.0
     */
    static CAPTURING_PHASE = 1;
    
    /**
     * 定义事件处于目标阶段。
     * @readonly
     * @type {Number}
     * @since 1.0.0
     */
    static AT_TARGET = 2;
    
    /**
     * 定义事件处于冒泡阶段。
     * @readonly
     * @type {Number}
     * @since 1.0.0
     */
    static BUBBLING_PHASE = 3;
}
